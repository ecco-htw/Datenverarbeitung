Welcome to the REST API of Ecco Project HTW!

Supported endpoints (all in JSON format):

  • GET /last_coordinates 
    {
  "data"  : Array of {
                           "id"          : String,
                           "coordinates" : {
                                             "longitude"   : number,
                                             "latitude"    : number
                                           }
                         }
    }
  • GET /measurements/{float_id}
     
    {
  "data"  : {
                  "saltinessValues"   : Array of     numbers,
    
              "pressureValues"    : Array of numbers,
    
              "temperatureValues" : Array of numbers,
    
              "path"              : Array of {
                                                   "longitude"   : number,
                                                   "latitude"    : number,
                                                   "cycleNumber" : number
                                                 }
                }
    }
  • GET /measurements/{float_id}/{cycle_num} 
    {
  "data"  : {
                  "saltinessValues"   : Array of     numbers,
    
              "pressureValues"    : Array of numbers,
    
              "temperatureValues" : Array of numbers
                }
    }
